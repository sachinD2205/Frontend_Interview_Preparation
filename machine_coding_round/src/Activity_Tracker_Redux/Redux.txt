import { createSlice, nanoid } from "@reduxjs/toolkit";

const activityTrackerSlice = createSlice({
  name: "activityTracker",
  initialState: {
    activities: []
  },
  reducers: {
    addActivity: {
      reducer(state, action) {
        state.activities.push(action.payload);
      }
    },
    deleteActivity: {
      reducer(state, action) {
        state.activities = state.activities.filter(
          (activity) => activity.id !== action.payload
        );
      }
    },
    editActivity: {
      reducer(state, action) {
        state.activities = state.activities.map((activity) =>
          activity.id === action.payload.id ? action.payload : activity
        );
      }
    }
  }
});

export const { addActivity, deleteActivity, editActivity } = activityTrackerSlice.actions;
export default activityTrackerSlice.reducer;

Scenario 1: "Your app is slow due to Redux re-renders. How do you fix it?"
Answer:
- Use useSelector with specific selectors (not entire state)
- Memoize selectors with useMemo or reselect
- Split large slices into smaller ones
- Use React.memo for components
- Check if you're creating new objects in selectors


Scenario 2: "When would you choose Context API over Redux?"
Answer:
For simple, infrequent state updates like:
- Theme switching
- User authentication status
- Language preference
Use Redux when state is complex, updates frequently, or needs DevTools.


Scenario 3: "How do you handle async operations in Redux?"
Answer:
- Use Redux Toolkit's createAsyncThunk
- Or use middleware like redux-thunk
- Never put async logic directly in reducers
